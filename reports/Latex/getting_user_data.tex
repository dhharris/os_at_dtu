\documentclass{article}
\usepackage[linesnumbered, ruled]{algorithm2e}

\SetKwProg{Fn}{Function}{}{}

\begin{document}
\begin{algorithm}[h] 
\caption{Getting and Executing Commands}
collection $\leftarrow$ initialize to empty Collection\;
status $\leftarrow$ false\;
line $\leftarrow$ NULL\;

\While{getline returns value $>$ 0}{
  remove newline character from $line$\;
  \ForEach{c in line}{
    status $\leftarrow$ parse\_character($Collection, c$)\;
    \If{status is true}{
      $break$\;
    }
  }
}
\end{algorithm}
\end{document}
